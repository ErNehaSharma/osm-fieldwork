166  def customTMS(self, url: str, name: str = "custom", source: str = "custom", suffix: str = "jpg"):
167  """Add a custom TMS URL to the list of sources.
168 
169  The url must end in %s to be replaced with the tile xyz values.
170 
171  Format examples:
172  https://basemap.nationalmap.gov/ArcGIS/rest/services/USGSTopo/MapServer/tile/{z}/{y}/{x}
173  https://maps.nyc.gov/xyz/1.0.0/carto/basemap/%s
174  https://maps.nyc.gov/xyz/1.0.0/carto/basemap/{z}/{x}/{y}.jpg
175 
176  The method will replace {z}/{x}/{y}.jpg with %s
177 
178  Args:
179  name (str): The name to display
180  url (str): The URL string
181  suffix (str): The suffix, png or jpg
182  source (str): The source value to use as an index
183  """
184  # Remove any file extensions if present and update the 'suffix' parameter
185  if url.endswith(".jpg"):
186  source = "jpg"
187  suffix = "jpg"
188  url = url[:-4] # Remove the last 4 characters (".jpg")
189  elif url.endswith(".png"):
190  source = "png"
191  suffix = "png"
192  url = url[:-4] # Remove the last 4 characters (".png")
193 
194  # Replace "{z}/{x}/{y}" with "%s"
195  url = re.sub(r"/{[xyz]+\}", "", url)
196  url = url + r"/%s"
197 
198  tms_params = {"name": name, "url": url, "suffix": suffix, "source": source}
199  log.debug(f"Setting custom TMS with params: {tms_params}")
200  self.sourcessources["custom"] = tms_params
201  self.sourcesource = "custom"

304  log.error(msg)

308  with open(boundary, "r") as f:
309  poly = geojson.load(f)
310  if "features" in poly:
311  geometry = shape(poly["features"][0]["geometry"])
312  elif "geometry" in poly:

315  geometry = shape(poly)

324  log.error(msg)

327  bbox = geometry.bounds
328  # left, bottom, right, top
329  # minX, minY, maxX, maxY

348  """Write PMTiles archive from tiles in the specified directory.
349 
350  Args:

356  Returns:
357  None

426  custom TMS if True.
427  outfile (str, optional): Output file name for the basemap.
428  zooms (str, optional): The Zoom levels, specified as a range
429  (e.g., "12-17") or comma-separated levels (e.g., "12,13,14").
430  outdir (str, optional): Output directory name for tile cache.
431  source (str, optional): Imagery source, one of
432  ["esri", "bing", "topo", "google", "oam"] (default is "esri").

434  Returns:
435  None
436  """
437  # if verbose, dump to the terminal.
438  if verbose is not None:
439  log.setLevel(logging.DEBUG)
440  ch = logging.StreamHandler(sys.stdout)
441  ch.setLevel(logging.DEBUG)

443  ch.setFormatter(formatter)
444  log.addHandler(ch)
445 
446  log.debug(
447  "Creating basemap with params: "
448  f"boundary={boundary} | "
449  f"outfile={outfile} | "
450  f"zooms={zooms} | "
451  f"outdir={outdir} | "

456

459  err = "You need to specify a boundary! (file or bbox)"
460  log.error(err)
461  raise ValueError(err)
462 
463  # Get all the zoom levels we want
464  zoom_levels = list()
465  if zooms:
466  if zooms.find("-") > 0:
467  start = int(zooms.split("-")[0])
468  end = int(zooms.split("-")[1]) + 1

473  levels = zooms.split(",")

475  zoom_levels.append(int(level))
476  else:
477  zoom_levels.append(int(zooms))

480  base = Path.cwd().absolute()
481  else:
482  base = Path(outdir).absolute()
483 
484  source = "custom" if tms else source
485  tiledir = base / f"{source}tiles"

487  tiledir.mkdir(parents=True, exist_ok=True)
488  # Convert to string for other methods
489  tiledir = str(tiledir)
490
