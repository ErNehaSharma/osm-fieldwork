59  (list): The submissions in JSON format
60  """

62  timer.start()
63  data = list()

80 class OdkCentral(object):
81  def __init__(
82  self,
83  url: Optional[str] = None,
84  user: Optional[str] = None,
85  passwd: Optional[str] = None,
86  ):
87  """A Class for accessing an ODK Central server via it's REST API.
88 
89  Args:
90  url (str): The URL of the ODK Central
91  user (str): The user's account name on ODK Central
92  passwd (str): The user's account password on ODK Central
93 
94  Returns:
95  (OdkCentral): An instance of this class
96  """
97  if not url:
98  url = os.getenv("ODK_CENTRAL_URL", default=None)
99  self.urlurl = url
100  if not user:
101  user = os.getenv("ODK_CENTRAL_USER", default=None)
102  self.useruser = user
103  if not passwd:
104  passwd = os.getenv("ODK_CENTRAL_PASSWD", default=None)
105  self.passwdpasswd = passwd
106  verify = os.getenv("ODK_CENTRAL_SECURE", default=True)
107  if type(verify) == str:
108  self.verifyverify = eval(verify)
109  else:
110  self.verifyverify = verify
111  # These are settings used by ODK Collect
112  self.generalgeneral = {
113  "form_update_mode": "match_exactly",
114  "autosend": "wifi_and_cellular",
115  }
116  # If there is a config file with authentication setting, use that
117  # so we don't have to supply this all the time. This is only used
118  # when odk_client is used, and no parameters are passed in.
119  if not self.urlurl:
120  # log.debug("Configuring ODKCentral from file .odkcentral")
121  home = os.getenv("HOME")
122  config = ".odkcentral"
123  filespec = home + "/" + config
124  if os.path.exists(filespec):
125  file = open(filespec, "r")
126  for line in file:
127  # Support embedded comments
128  if line[0] == "#":
129  continue
130  # Read the config file for authentication settings
131  tmp = line.split("=")
132  if tmp[0] == "url":
133  self.urlurl = tmp[1].strip("\n")
134  if tmp[0] == "user":
135  self.useruser = tmp[1].strip("\n")
136  if tmp[0] == "passwd":
137  self.passwdpasswd = tmp[1].strip("\n")
138  else:
139  log.warning(f"Authentication settings missing from {filespec}")
140  else:
141  log.debug(f"ODKCentral configuration parsed: {self.url}")
142  # Base URL for the REST API
143  self.versionversion = "v1"
144  # log.debug(f"Using {self.version} API")
145  self.basebase = self.urlurl + "/" + self.versionversion + "/"
146 
147  # Use a persistant connect, better for multiple requests
148  self.sessionsession = requests.Session()
149 
150  # Authentication with session token
151  self.authenticateauthenticate()
152 
153  # These are just cached data from the queries
154  self.projectsprojects = dict()
155  self.usersusers = list()
156  # The number of threads is based on the CPU cores
157  info = get_cpu_info()
158  self.corescores = info["count"]

160  def authenticate(
161  self,
162  url: str = None,
163  user: str = None,
164  passwd: str = None,
165  ):
166  """Setup authenticate to an ODK Central server.
167 
168  Args:
169  url (str): The URL of the ODK Central
170  user (str): The user's account name on ODK Central
171  passwd (str): The user's account password on ODK Central
172 
173  Returns:
174  (requests.Response): A response from ODK Central after auth
175  """
176  if not self.urlurl:
177  self.urlurl = url
178  if not self.useruser:
179  self.useruser = user
180  if not self.passwdpasswd:
181  self.passwdpasswd = passwd
182  # Enable persistent connection, create a cookie for this session
183  self.sessionsession.headers.update({"accept": "odkcentral"})
184 
185  # Get a session token
186  response = self.sessionsession.post(
187  f"{self.base}sessions",
188  json={
189  "email": self.useruser,
190  "password": self.passwdpasswd,
191  },
192  )
193  if not response.ok:
194  response.raise_for_status()
195 
196  self.sessionsession.headers.update({"Authorization": f"Bearer {response.json().get('token')}"})
197 
198  # Connect to the server
199  return self.sessionsession.get(self.urlurl, verify=self.verifyverify)

415  def getAllSubmissions(self, project_id: int, xforms: list = None, filters: dict = None):
416  """Fetch a list of submissions in a project on an ODK Central server.
417 
418  Args:
419  project_id (int): The ID of the project on ODK Central
420  xforms (list): The list of XForms to get the submissions of
421 
422  Returns:
423  (json): All of the submissions for all of the XForm in a project
424  """
425  timer = Timer(text="getAllSubmissions() took {seconds:.0f}s")
426  timer.start()
427  if not xforms:
428  xforms_data = self.listFormslistForms(project_id)
429  xforms = [d["xmlFormId"] for d in xforms_data]
430 
431  chunk = round(len(xforms) / self.corescores) if round(len(xforms) / self.corescores) > 0 else 1
432  last_slice = len(xforms) if len(xforms) % chunk == 0 else len(xforms) - 1
433  cycle = range(0, (last_slice + chunk) + 1, chunk)
434  future = None
435  result = None
436  previous = 0
437  newdata = list()
438 
439  # single threaded for easier debugging
440  # for current in cycle:
441  # if previous == current:
442  # continue
443  # result = downloadThread(project_id, xforms[previous:current])
444  # previous = current
445  # newdata += result
446 
447  odk_credentials = {"url": self.urlurl, "user": self.useruser, "passwd": self.passwdpasswd}
448 
449  with concurrent.futures.ThreadPoolExecutor(max_workers=self.corescores) as executor:
450  futures = list()
451  for current in cycle:
452  if previous == current:
453  continue
454  result = executor.submit(downloadThread, project_id, xforms[previous:current], odk_credentials, filters)
455  previous = current
456  futures.append(result)
457  for future in concurrent.futures.as_completed(futures):
458  log.debug("Waiting for thread to complete..")
459  data = future.result(timeout=10)
460  if len(data) > 0:
461  newdata += data
462  timer.stop()
463  return newdata
